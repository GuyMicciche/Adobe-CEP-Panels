1. Change port in cep.config.ts file to 8080:

const config: CEP_Config = {
  // ... your existing config
  version,
  id: "com.bolt.cep", 
  displayName: "Bolt CEP", 
  symlink: "local",
  port: 8080,
  servePort: 5000,
  startingDebugPort: 8860,
  extensionManifestVersion: 6.0,
  requiredRuntimeVersion: 9.0,
  // ... rest of config


2. Add this to your vite.config.ts file:


export default defineConfig({
  // ... your existing config
  server: {
    host: '0.0.0.0',
    port: cepConfig.port,
    watch: {
      usePolling: true,
      interval: 1000,
    },
  },
  preview: {
    host: '0.0.0.0',
    port: cepConfig.servePort,
  },
  // ... rest of config
});


3. Run put create-symlink.bat inside your Bolt-CEP product directory after you create the project and run it to create the symlink.
